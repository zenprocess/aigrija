function buildPage(opts: { lang: Lang; path: string; title: string; content: string }): string {
  const { lang, path, title, content } = opts;
  const base = 'https://ai-grija.ro';
  const hreflangs = LANGS.map(l => `  <link rel="alternate" hreflang="${l}" href="${base}${path}?lang=${l}">`).join('\n');
  const disclaimerHtml = lang !== 'ro' ? `<div class="disclaimer">\u26a0\ufe0f ${disclaimers[lang as Exclude<Lang, 'ro'>]}</div>` : '';
  const langSwitcher = LANGS.map(l => {
    const cls = l === lang ? ' class="active"' : '';
    return `<a href="${path}?lang=${l}"${cls}>${langLabels[l]}</a>`;
  }).join(' ');
  return `<!DOCTYPE html>
<html lang="${lang}">
<head>
  <meta charset="utf-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>${title}</title>
  <link rel="canonical" href="${base}${path}">
${hreflangs}
  <style>${CSS}</style>
</head>
<body>
  <div class="container">
    <header>
      <a class="brand" href="https://ai-grija.ro">ai-grija.ro</a>
      <nav class="lang-switch">${langSwitcher}</nav>
    </header>
    ${disclaimerHtml}
    ${content}
    <a class="back" href="https://ai-grija.ro">\u2190 ai-grija.ro</a>
  </div>
</body>
</html>`;
}
